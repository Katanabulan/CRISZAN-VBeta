4. Kondisi (if/else)

If Statement

Menjalankan kode jika kondisi benar (true). 

let umur = 18;

if (umur >= 18) {
    console.log('Anda sudah dewasa');
}

If...Else

Menjalankan kode berbeda berdasarkan kondisi. 

let nilai = 75;

if (nilai >= 70) {
    console.log('Anda lulus');
} else {
    console.log('Anda tidak lulus');
}

Else If

Untuk multiple conditions. 

let nilai = 85;

if (nilai >= 90) {
    console.log('Grade: A');
} else if (nilai >= 80) {
    console.log('Grade: B');
} else if (nilai >= 70) {
    console.log('Grade: C');
} else {
    console.log('Grade: D');
}

Switch Case

Untuk banyak kondisi yang membandingkan nilai yang sama. 

let hari = 'Senin';

switch (hari) {
    case 'Senin':
        console.log('Hari kerja');
        break;
    case 'Sabtu':
    case 'Minggu':
        console.log('Hari libur');
        break;
    default:
        console.log('Hari kerja');
}

Ternary Operator

Shortcut untuk if...else sederhana. 

let umur = 20;
let status = umur >= 18 ? 'Dewasa' : 'Anak-anak';
console.log(status); // Dewasa

// Sama dengan:
if (umur >= 18) {
    status = 'Dewasa';
} else {
    status = 'Anak-anak';
}

5. Perulangan (Loops)

For Loop

Untuk perulangan dengan jumlah yang diketahui. 

// Menampilkan angka 1-5
for (let i = 1; i <= 5; i++) {
    console.log(i);
}

// Hasil: 1, 2, 3, 4, 5

While Loop

Untuk perulangan selama kondisi true. 

let i = 1;
while (i <= 5) {
    console.log(i);
    i++;
}

Do...While Loop

Sama seperti while, tapi menjalankan minimal sekali. 

let i = 1;
do {
    console.log(i);
    i++;
} while (i <= 5);

Loop dengan Array

let fruits = ['Apel', 'Jeruk', 'Mangga'];

// For loop
for (let i = 0; i < fruits.length; i++) {
    console.log(fruits[i]);
}

// For...of (modern)
for (let fruit of fruits) {
    console.log(fruit);
}

// forEach method
fruits.forEach(function(fruit) {
    console.log(fruit);
});

6. Fungsi (Functions)

Apa itu Fungsi?

Blok kode yang dapat dipanggil berulang kali.

Function Declaration

// Membuat fungsi
function sapa(nama) {
    return 'Halo, ' + nama + '!';
}

// Memanggil fungsi
let pesan = sapa('Budi');
console.log(pesan); // Halo, Budi!

Function Expression

const sapa = function(nama) {
    return 'Halo, ' + nama + '!';
};

console.log(sapa('Ani')); // Halo, Ani!

Arrow Function (Modern)

// Bentuk sederhana
const sapa = (nama) => 'Halo, ' + nama + '!';

// Bentuk dengan multiple lines
const hitungLuas = (panjang, lebar) => {
    let luas = panjang * lebar;
    return luas;
};

console.log(sapa('Citra')); // Halo, Citra!
console.log(hitungLuas(5, 3)); // 15

Parameter dan Return

// Fungsi dengan multiple parameters
function hitungTotal(harga, jumlah) {
    return harga * jumlah;
}

// Fungsi tanpa return
function tampilkanPesan(nama) {
    console.log('Selamat datang, ' + nama);
}

let total = hitungTotal(10000, 3);
console.log(total); // 30000

tampilkanPesan('Doni'); // Selamat datang, Doni

7. Array

Apa itu Array?

Struktur data untuk menyimpan kumpulan nilai.

Membuat Array

// Cara 1: Array literal
let fruits = ['Apel', 'Jeruk', 'Mangga'];

// Cara 2: Constructor
let numbers = new Array(1, 2, 3, 4, 5);

// Array campuran
let mixed = ['Teks', 123, true, null];

Mengakses Array

let fruits = ['Apel', 'Jeruk', 'Mangga'];

console.log(fruits[0]); // Apel (index 0)
console.log(fruits[1]); // Jeruk (index 1)
console.log(fruits[2]); // Mangga (index 2)
console.log(fruits.length); // 3 (panjang array)

Method Array Penting

let fruits = ['Apel', 'Jeruk'];

// push - tambah di akhir
fruits.push('Mangga');
console.log(fruits); // ['Apel', 'Jeruk', 'Mangga']

// pop - hapus dari akhir
let last = fruits.pop();
console.log(last); // Mangga

// unshift - tambah di awal
fruits.unshift('Pisang');
console.log(fruits); // ['Pisang', 'Apel', 'Jeruk']

// shift - hapus dari awal
let first = fruits.shift();
console.log(first); // Pisang

// indexOf - cari index
let index = fruits.indexOf('Jeruk');
console.log(index); // 1

// slice - potong array
let someFruits = fruits.slice(0, 2);
console.log(someFruits); // ['Apel', 'Jeruk']

8. Object

Apa itu Object?

Struktur data untuk menyimpan data terstruktur dalam key-value pairs.

Membuat Object

// Object literal
let person = {
    nama: 'Budi',
    umur: 25,
    pekerjaan: 'Developer',
    isActive: true
};

Mengakses Object

let person = {
    nama: 'Budi',
    umur: 25
};

// Dot notation
console.log(person.nama); // Budi
console.log(person.umur); // 25

// Bracket notation
console.log(person['nama']); // Budi
console.log(person['umur']); // 25

Memodifikasi Object

let person = {
    nama: 'Budi',
    umur: 25
};

// Menambah property
person.pekerjaan = 'Developer';

// Mengubah property
person.umur = 26;

// Menghapus property
delete person.pekerjaan;

console.log(person);

Method dalam Object

let person = {
    nama: 'Budi',
    umur: 25,
    sapa: function() {
        return 'Halo, nama saya ' + this.nama;
    },
    // Arrow function (tidak bisa pakai this)
    perkenalan: () => {
        return 'Halo!';
    }
};

console.log(person.sapa()); // Halo, nama saya Budi

9. DOM Manipulation

Apa itu DOM?

DOM (Document Object Model) adalah representasi object dari HTML yang memungkinkan JavaScript mengakses dan memanipulasi elemen HTML.

Seleksi Elemen

// By ID
let header = document.getElementById('header');

// By Class (returns array-like)
let items = document.getElementsByClassName('item');

// By Tag Name
let paragraphs = document.getElementsByTagName('p');

// Modern selectors (CSS-like)
let element = document.querySelector('#header');
let allItems = document.querySelectorAll('.item');

Mengubah Konten

let title = document.getElementById('title');

// Mengubah teks
title.textContent = 'Judul Baru';

// Mengubah HTML
title.innerHTML = '<strong>Judul Baru</strong>';

// Mengubah style
title.style.color = 'red';
title.style.fontSize = '24px';

Event Handling

let button = document.getElementById('myButton');

// Cara 1: Property
button.onclick = function() {
    alert('Button diklik!');
};

// Cara 2: addEventListener (recommended)
button.addEventListener('click', function() {
    alert('Button diklik!');
});

// Event lainnya: mouseover, keypress, submit, dll

Contoh Praktis

<button id="btn">Klik Saya</button>
<p id="output"></p>

<script>
    let btn = document.getElementById('btn');
    let output = document.getElementById('output');
    
    btn.addEventListener('click', function() {
        output.textContent = 'Button telah diklik!';
        output.style.color = 'green';
    });
</script>
